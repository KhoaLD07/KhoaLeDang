void go(char* buff, int len) {
	datap data;
	WIN32_FIND_DATAA file_data;
	HANDLE file_handle = NULL;
	formatp out_buffer;
	char* target_dir = NULL;
	int length = 0;
	BeaconDataParse(&data, buff, len);
	
	target_dir = BeaconDataExtract(&data, &length);
	int out_size = 0;
	BeaconFormatAlloc(&out_buffer, 8192);
	file_handle = FindFirstFileA(target_dir, &file_data);
	if (file_handle == INVALID_HANDLE_VALUE)
	{
		BeaconPrintf(CALLBACK_ERROR, "FAILED TO FIND FIRST FILE\n");
		BeaconFormatFree(&out_buffer);
		return;
	}
	BeaconFormatPrintf(&out_buffer, "---List---\n");
	do
	{
		BeaconFormatPrintf(&out_buffer, "%s\n", file_data.cFileName);
	} while (FindNextFileA(file_handle, &file_data) != 0);

	char* final_output = BeaconFormatToString(&out_buffer, &out_size);
	BeaconPrintf(CALLBACK_OUTPUT, "%s\n", final_output);
	FindClose(file_handle);
	BeaconFormatFree(&out_buffer);
	return;
}

alias -ls {
		local('$barch $handle $data $args $target_dir $bof_path');
	
		if (size(@_) != 2) {
			berror($1, "How to use: -ls [path]");
			return;
		}
		
		$target_dir = $2;
	
		if (strlen($target_dir) == 0) {
			berror($1, "Path cannot be empty");
			return;
		}
	
		if (![$target_dir endsWith: "*"]) {
			if ([$target_dir endsWith: "\\"] || [$target_dir endsWith: "/"]) {
				$target_dir = $target_dir . "*";
			} else {
				$target_dir = $target_dir . "\\*";
			}
		}
	
		$barch = barch($1);
		$bof_path = script_resource("BOF2." . $barch . ".obj");
	
		if (!-exists $bof_path) {
			berror($1, "BOF not found: " . $bof_path);
			return;
		}
	
		$handle = openf($bof_path);
		$data   = readb($handle, -1);
		closef($handle);
	
		$args = bof_pack($1, "z", $target_dir);
	
		btask($1, "Sending to Beacon... Path: " . $target_dir);
		beacon_inline_execute($1, $data, "go", $args);
	}
